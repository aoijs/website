import Example from "../../../../components/Page/Example.astro";
import { Aside } from '@astrojs/starlight/components';

<Aside title="Intents">

This requires the `GuildWebhooks` intent!

</Aside>

<Example>

<div slot="default">

```js
client.webhooksUpdateCommand({
    name: "Webhooks Update",
    channel: "$channelId",
    code: `
        $log[Webhook updated!]
    `
});
```

</div>
<div slot="handler">

```js
module.exports = {
    name: "Webhooks Update",
    channel: "$channelId",
    type: "webhooksUpdate",
    code: `
        $log[Webhook updated!]
    `
};
```

</div>
</Example>
