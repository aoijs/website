import Example from "../../../../components/Page/Example.astro";
import { Aside } from '@astrojs/starlight/components';

<Aside title="Intents">

This requires the `GuildMessageReactions` intent!

</Aside>

<Aside title="Functions" type="tip">

This event has a function to retrieve data:

[$reactionData[option]](/functions/reactiondata)

</Aside>

<Example>

<div slot="default">

```js
client.reactionRemoveCommand({
    name: "Reaction Remove",
    channel: "$channelId",
    code: `
        $log[$authorId just reacted with $reactionData[emoji] on $messagId!]
    `
});
```

</div>
<div slot="handler">

```js
module.exports = {
    name: "Reaction Remove",
    channel: "$channelId",
    type: "reactionRemove",
    code: `
        $log[$authorId just reacted with $reactionData[emoji] on $messageId!]
    `
};
```

</div>
</Example>
