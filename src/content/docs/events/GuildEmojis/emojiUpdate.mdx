import Example from "../../../../components/Page/Example.astro";
import { Aside } from '@astrojs/starlight/components';

<Aside title="Intents">

This requires the `GuildEmojis` intent!

</Aside>

<Aside title="Functions" type="tip">

This event has a function to retrieve data:

[$newEmoji[option]](/functions/newemoji)

[$oldEmoji[option]](/functions/oldemoji)

</Aside>

<Example>

<div slot="default">

```js
client.emojiUpdateCommand({
    name: "Guild Emoji Update",
    channel: "$channelId",
    code: `
        $log[A new emoji was updated with the name $oldEmoji[string]!]
    `
});
```

</div>
<div slot="handler">

```js
module.exports = {
    name: "Guild Emoji Update",
    channel: "$channelId",
    type: "emojiUpdate",
    code: `
        $log[A new emoji was updated with the name $oldEmoji[string]!]
    `
};
```

</div>
</Example>
