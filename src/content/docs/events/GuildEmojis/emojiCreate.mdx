import Example from "../../../../components/Page/Example.astro";
import { Aside } from '@astrojs/starlight/components';

<Aside title="Intents">

This requires the `GuildEmojisAndStickers` intent!

</Aside>

<Aside title="Functions" type="tip">

This event has a function to retrieve data:

[$newEmoji[option]](/functions/newemoji)

[$oldEmoji[option]](/functions/oldemoji)

</Aside>

<Example>

<div slot="default">

```js
client.emojiCreateCommand({
    name: "Guild Emoji Create",
    channel: "$channelId",
    code: `
        $log[A new emoji was created with the name $newEmoji[string]!]
    `
});
```

</div>
<div slot="handler">

```js
module.exports = {
    name: "Guild Emoji Create",
    channel: "$channelId",
    type: "emojiCreate",
    code: `
        $log[A new emoji was created with the name $newEmoji[string]!]
    `
};
```

</div>
</Example>
